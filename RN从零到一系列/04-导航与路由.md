## 📊 导航方案对比

| 方案 | 特点 | 推荐度 |
|------|------|--------|
| Expo Router | 基于文件路由，类似 Next.js | ⭐⭐⭐⭐⭐（Expo 项目首选） |

## 🗂 1. Expo Router — 文件路由

### 目录即路由

```
app/
├── _layout.tsx           # 根布局
├── index.tsx             # 首页 → /
├── about.tsx             # 关于页 → /about
├── settings.tsx          # 设置 → /settings
├── (tabs)/               # Tab 导航组
│   ├── _layout.tsx       # Tab 布局
│   ├── index.tsx         # Tab 1：首页
│   ├── explore.tsx       # Tab 2：发现
│   └── profile.tsx       # Tab 3：我的
├── user/
│   ├── [id].tsx          # 动态路由 → /user/123
│   └── settings.tsx      # 嵌套路由 → /user/settings
└── (auth)/               # 分组（不影响 URL）
    ├── login.tsx         # → /login
    └── register.tsx      # → /register
```

### 根布局

```tsx
// app/_layout.tsx
import { Stack } from 'expo-router';

export default function RootLayout() {
  return (
    <Stack
      screenOptions={{
        headerStyle: { backgroundColor: '#6366F1' },
        headerTintColor: '#fff',
        headerTitleStyle: { fontWeight: 'bold' },
      }}
    >
      <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
      <Stack.Screen name="about" options={{ title: '关于我们' }} />
    </Stack>
  );
}
```

### Tab 导航布局

```tsx
// app/(tabs)/_layout.tsx
import { Tabs } from 'expo-router';
import { Ionicons } from '@expo/vector-icons';

export default function TabLayout() {
  return (
    <Tabs
      screenOptions={{
        tabBarActiveTintColor: '#6366F1',
        tabBarInactiveTintColor: '#9CA3AF',
        tabBarStyle: {
          backgroundColor: '#fff',
          borderTopColor: '#F3F4F6',
        },
        headerStyle: { backgroundColor: '#6366F1' },
        headerTintColor: '#fff',
      }}
    >
      <Tabs.Screen
        name="index"
        options={{
          title: '首页',

            <Ionicons name="home" size={size} color={color} />
          ),
        }}
      />
      <Tabs.Screen
        name="explore"
        options={{
          title: '发现',

            <Ionicons name="compass" size={size} color={color} />
          ),
        }}
      />
      <Tabs.Screen
        name="profile"
        options={{
          title: '我的',
          tabBarIcon: ({ color, size }) => (
            <Ionicons name="person" size={size} color={color} />
          ),
        }}
      />
    </Tabs>
  );
}
```

### 页面跳转

```tsx
import { Link, router } from 'expo-router';

// 方式一：Link 组件（声明式）
<Link href="/about">
  <Text>关于我们</Text>
</Link>

// 带参数
<Link href="/user/123">
  <Text>用户详情</Text>
</Link>

// 方式二：router API（命令式）
router.push('/about');              // 跳转（可返回）
router.replace('/login');           // 替换（不可返回）
router.back();                     // 返回上一页
router.navigate('/explore');        // 导航（已在栈中则回到该页）

// 带参数跳转
router.push({
  pathname: '/user/[id]',
  params: { id: '123', name: '张三' },
});
```

### 接收参数

```tsx
// app/user/[id].tsx
import { useLocalSearchParams } from 'expo-router';

export default function UserDetail() {
  const { id, name } = useLocalSearchParams<{ id: string; name?: string }>();

  return (
    <View>
      <Text>用户 ID: {id}</Text>
      <Text>用户名: {name}</Text>
    </View>
  );
}
```

## 🧭 2. React Navigation — 配置式路由

如果不使用 Expo Router，可以用 React Navigation：

### 安装

```bash
npm install @react-navigation/native
npm install @react-navigation/native-stack
npm install @react-navigation/bottom-tabs
npx expo install react-native-screens react-native-safe-area-context
```

### Stack 导航

```tsx
import { NavigationContainer } from '@react-navigation/native';
import { createNativeStackNavigator } from '@react-navigation/native-stack';

// 定义路由参数类型
type RootStackParamList = {
  Home: undefined;
  Detail: { id: string; title: string };
  Settings: undefined;
};

const Stack = createNativeStackNavigator<RootStackParamList>();

export default function App() {
  return (
    <NavigationContainer>
      <Stack.Navigator
        initialRouteName="Home"
        screenOptions={{
          headerStyle: { backgroundColor: '#6366F1' },
          headerTintColor: '#fff',
        }}
      >
        <Stack.Screen
          name="Home"
          component={HomeScreen}
          options={{ title: '首页' }}
        />
        <Stack.Screen
          name="Detail"
          component={DetailScreen}
          options={({ route }) => ({ title: route.params.title })}
        />
        <Stack.Screen name="Settings" component={SettingsScreen} />
      </Stack.Navigator>
    </NavigationContainer>
  );
}
```

### 页面跳转与参数

```tsx
import { NativeStackScreenProps } from '@react-navigation/native-stack';

type HomeProps = NativeStackScreenProps<RootStackParamList, 'Home'>;

function HomeScreen({ navigation }: HomeProps) {
  return (
    <View>
      {/* 跳转并传参 */}
      <Pressable
        onPress={() => navigation.navigate('Detail', { id: '1', title: '文章标题' })}
      >
        <Text>查看详情</Text>
      </Pressable>

      {/* 替换当前页 */}
      <Pressable onPress={() => navigation.replace('Settings')}>
        <Text>设置</Text>
      </Pressable>

      {/* 返回 */}
      <Pressable onPress={() => navigation.goBack()}>
        <Text>返回</Text>
      </Pressable>
    </View>
  );
}

// 接收参数
type DetailProps = NativeStackScreenProps<RootStackParamList, 'Detail'>;

function DetailScreen({ route, navigation }: DetailProps) {
  const { id, title } = route.params;

  return (
    <View>
      <Text>ID: {id}</Text>
      <Text>标题: {title}</Text>
    </View>
  );
}
```

### Tab 导航

```tsx
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';

const Tab = createBottomTabNavigator();

function MainTabs() {
  return (
    <Tab.Navigator
      screenOptions={({ route }) => ({
        tabBarIcon: ({ color, size }) => {
          const icons: Record<string, string> = {
            Home: 'home',
            Explore: 'compass',
            Profile: 'person',
          };
          return <Ionicons name={icons[route.name]} size={size} color={color} />;
        },
        tabBarActiveTintColor: '#6366F1',
      })}
    >
      <Tab.Screen name="Home" component={HomeScreen} options={{ title: '首页' }} />
      <Tab.Screen name="Explore" component={ExploreScreen} options={{ title: '发现' }} />
      <Tab.Screen name="Profile" component={ProfileScreen} options={{ title: '我的' }} />
    </Tab.Navigator>
  );
}
```

## 🔀 3. 常见导航模式

### 嵌套导航（Tab + Stack）

```
NavigationContainer
└── Stack.Navigator（根）
    ├── MainTabs（Tab Navigator）
    │   ├── Home → HomeStack
    │   ├── Explore
    │   └── Profile
    ├── Settings（全屏页面）
    ├── Detail（全屏页面）
    └── Login（Modal 页面）
```

### Modal 弹窗页面

```tsx
// Expo Router
<Stack.Screen
  name="modal"
  options={{ presentation: 'modal' }}
/>

// React Navigation
<Stack.Screen
  name="Modal"
  component={ModalScreen}
  options={{ presentation: 'modal' }}
/>
```

### 自定义返回行为

```tsx
import { useNavigation } from '@react-navigation/native';

function CustomBackButton() {
  const navigation = useNavigation();

  useEffect(() => {
    navigation.setOptions({
      headerLeft: () => (
        <Pressable onPress={() => {
          Alert.alert('确定要离开吗？', '', [
            { text: '取消' },
            { text: '确定', onPress: () => navigation.goBack() },
          ]);
        }}>
          <Text>← 返回</Text>
        </Pressable>
      ),
    });
  }, [navigation]);
}
```

## 🔐 4. 登录拦截

```tsx
// app/_layout.tsx（Expo Router 示例）
import { Redirect, Stack } from 'expo-router';

export default function RootLayout() {
  const { isLoggedIn } = useAuth();

  // 未登录时重定向到登录页
  if (!isLoggedIn) {
    return <Redirect href="/login" />;
  }

  return (
    <Stack>
      <Stack.Screen name="(tabs)" options={{ headerShown: false }} />
    </Stack>
  );
}
```
