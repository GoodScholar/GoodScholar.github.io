## 🎬 1. Animated — 内置动画

### 基础动画类型

```tsx
import { Animated, Easing } from 'react-native';

export default function AnimationDemo() {
  // 创建动画值
  const fadeAnim = useRef(new Animated.Value(0)).current;
  const slideAnim = useRef(new Animated.Value(-100)).current;
  const scaleAnim = useRef(new Animated.Value(0.5)).current;

  const startAnimation = () => {
    // 🔹 渐隐渐显
    Animated.timing(fadeAnim, {
      toValue: 1,
      duration: 500,
      useNativeDriver: true,  // ⚠️ 开启原生驱动（性能更好）
    }).start();

    // 🔹 弹簧动画
    Animated.spring(slideAnim, {
      toValue: 0,
      friction: 5,        // 摩擦力（越大停得越快）
      tension: 80,        // 张力（越大弹得越快）
      useNativeDriver: true,
    }).start();

    // 🔹 缓动动画
    Animated.timing(scaleAnim, {
      toValue: 1,
      duration: 600,
      easing: Easing.bounce,  // 弹跳效果
      useNativeDriver: true,
    }).start();
  };

  return (
    <View style={styles.container}>
      <Animated.View style={[styles.box, { opacity: fadeAnim }]}>
        <Text>渐显</Text>
      </Animated.View>

      <Animated.View style={[styles.box, { transform: [{ translateY: slideAnim }] }]}>
        <Text>滑入</Text>
      </Animated.View>

      <Animated.View style={[styles.box, { transform: [{ scale: scaleAnim }] }]}>
        <Text>缩放</Text>
      </Animated.View>

      <Pressable onPress={startAnimation}>
        <Text>播放动画</Text>
      </Pressable>
    </View>
  );
}
```

### 组合动画

```tsx
// 同时播放
Animated.parallel([
  Animated.timing(fadeAnim, { toValue: 1, duration: 300, useNativeDriver: true }),
  Animated.spring(scaleAnim, { toValue: 1, useNativeDriver: true }),
]).start();

// 顺序播放
Animated.sequence([
  Animated.timing(fadeAnim, { toValue: 1, duration: 200, useNativeDriver: true }),
  Animated.delay(100),
  Animated.spring(slideAnim, { toValue: 0, useNativeDriver: true }),
]).start();

// 交错播放（列表项依次出现）
Animated.stagger(100, [
  Animated.timing(item1Anim, { toValue: 1, duration: 300, useNativeDriver: true }),
  Animated.timing(item2Anim, { toValue: 1, duration: 300, useNativeDriver: true }),
  Animated.timing(item3Anim, { toValue: 1, duration: 300, useNativeDriver: true }),
]).start();

// 循环动画
Animated.loop(
  Animated.sequence([
    Animated.timing(rotateAnim, {
      toValue: 1,
      duration: 2000,
      easing: Easing.linear,
      useNativeDriver: true,
    }),
  ])
).start();
```

### 插值映射

```tsx
const rotateAnim = useRef(new Animated.Value(0)).current;

// 数值 0-1 映射为旋转 0°-360°
const spin = rotateAnim.interpolate({
  inputRange: [0, 1],
  outputRange: ['0deg', '360deg'],
});

// 滚动位置映射为透明度
const scrollY = useRef(new Animated.Value(0)).current;
const headerOpacity = scrollY.interpolate({
  inputRange: [0, 150],
  outputRange: [0, 1],
  extrapolate: 'clamp',  // 超出范围不外推
});

<Animated.View style={{ transform: [{ rotate: spin }] }}>
  <Text>🔄 旋转中...</Text>
</Animated.View>
```

## ✨ 2. LayoutAnimation — 布局动画

一行代码让布局变化自带动画效果：

```tsx
import { LayoutAnimation, Platform, UIManager } from 'react-native';

// Android 需要开启
if (Platform.OS === 'android') {
  UIManager.setLayoutAnimationEnabledExperimental?.(true);
}

function ExpandableCard() {
  const [expanded, setExpanded] = useState(false);

  const toggleExpand = () => {
    // 在 setState 之前调用
    LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut);
    setExpanded(!expanded);
  };

  return (
    <Pressable onPress={toggleExpand} style={styles.card}>
      <Text style={styles.title}>点击展开/收起</Text>
      {expanded && (
        <Text style={styles.content}>
          这里是展开后的详细内容，LayoutAnimation 自动添加了
          展开和收起的过渡动画，无需手动管理 Animated.Value。
        </Text>
      )}
    </Pressable>
  );
}
```

## 🚀 3. Reanimated — 高性能动画

Reanimated 在 UI 线程运行动画，避免 JS 线程卡顿，实现真正的 60fps：

### 安装

```bash
npx expo install react-native-reanimated
```

### 基础用法

```tsx
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withSpring,
  withTiming,
  withRepeat,
  withSequence,
  Easing,
} from 'react-native-reanimated';

export default function ReanimatedDemo() {
  const offset = useSharedValue(0);
  const scale = useSharedValue(1);
  const rotation = useSharedValue(0);

  // 动画样式（在 UI 线程计算）
  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateX: offset.value },
      { scale: scale.value },
      { rotate: `${rotation.value}deg` },
    ],
  }));

  const handlePress = () => {
    // 弹簧动画
    offset.value = withSpring(offset.value === 0 ? 100 : 0, {
      damping: 15,
      stiffness: 100,
    });

    // 缩放
    scale.value = withSequence(
      withTiming(1.2, { duration: 150 }),
      withSpring(1)
    );
  };

  // 无限旋转
  const startRotation = () => {
    rotation.value = withRepeat(
      withTiming(360, { duration: 2000, easing: Easing.linear }),
      -1,  // -1 = 无限循环
      false
    );
  };

  return (
    <View style={styles.container}>
      <Animated.View style={[styles.box, animatedStyle]}>
        <Text>🎯</Text>
      </Animated.View>
      <Pressable onPress={handlePress}>
        <Text>弹跳</Text>
      </Pressable>
      <Pressable onPress={startRotation}>
        <Text>旋转</Text>
      </Pressable>
    </View>
  );
}
```

### 进入/退出动画

```tsx
import Animated, { FadeIn, FadeOut, SlideInRight, BounceIn } from 'react-native-reanimated';

function AnimatedList({ items }) {
  return (
    <View>
      {items.map((item, index) => (
        <Animated.View
          key={item.id}
          entering={SlideInRight.delay(index * 100).springify()}
          exiting={FadeOut.duration(200)}
        >
          <Text>{item.title}</Text>
        </Animated.View>
      ))}
    </View>
  );
}

// 常用预设动画
// 进入: FadeIn, SlideInRight, SlideInLeft, SlideInUp, BounceIn, ZoomIn, FlipInX
// 退出: FadeOut, SlideOutRight, SlideOutLeft, BounceOut, ZoomOut, FlipOutX
```

## 👆 4. 手势处理

### 安装

```bash
npx expo install react-native-gesture-handler
```

### 基础手势

```tsx
import { GestureDetector, Gesture } from 'react-native-gesture-handler';
import Animated, { useSharedValue, useAnimatedStyle, withSpring } from 'react-native-reanimated';

export default function GestureDemo() {
  const translateX = useSharedValue(0);
  const translateY = useSharedValue(0);
  const scale = useSharedValue(1);

  // 拖拽手势
  const panGesture = Gesture.Pan()
    .onUpdate((e) => {
      translateX.value = e.translationX;
      translateY.value = e.translationY;
    })
    .onEnd(() => {
      translateX.value = withSpring(0);
      translateY.value = withSpring(0);
    });

  // 缩放手势
  const pinchGesture = Gesture.Pinch()
    .onUpdate((e) => {
      scale.value = e.scale;
    })
    .onEnd(() => {
      scale.value = withSpring(1);
    });

  // 组合手势（同时响应）
  const composedGesture = Gesture.Simultaneous(panGesture, pinchGesture);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateX: translateX.value },
      { translateY: translateY.value },
      { scale: scale.value },
    ],
  }));

  return (
    <GestureDetector gesture={composedGesture}>
      <Animated.View style={[styles.box, animatedStyle]}>
        <Text>拖我/捏我</Text>
      </Animated.View>
    </GestureDetector>
  );
}
```

### 实战：下拉刷新卡片

```tsx
function PullToRefreshCard() {
  const translateY = useSharedValue(0);
  const [refreshing, setRefreshing] = useState(false);

  const panGesture = Gesture.Pan()
    .onUpdate((e) => {
      if (e.translationY > 0) {
        translateY.value = Math.min(e.translationY * 0.5, 100);
      }
    })
    .onEnd(() => {
      if (translateY.value > 60) {
        // 触发刷新
        runOnJS(setRefreshing)(true);
      }
      translateY.value = withSpring(0);
    });

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateY: translateY.value }],
  }));

  return (
    <GestureDetector gesture={panGesture}>
      <Animated.View style={[styles.card, animatedStyle]}>
        <Text>下拉刷新</Text>
      </Animated.View>
    </GestureDetector>
  );
}
```
