## 🎣 1. useState — 组件内状态

```tsx
import { useState } from 'react';
import { View, Text, Pressable, TextInput, StyleSheet } from 'react-native';

export default function Counter() {
  // 基础状态
  const [count, setCount] = useState(0);
  // 对象状态
  const [user, setUser] = useState({ name: '', age: 0 });
  // 数组状态
  const [todos, setTodos] = useState<string[]>([]);
  const [input, setInput] = useState('');

  // 添加 todo
  const addTodo = () => {
    if (!input.trim()) return;
    setTodos(prev => [...prev, input]);  // ⚠️ 不要直接 push，要创建新数组
    setInput('');
  };

  // 删除 todo
  const removeTodo = (index: number) => {
    setTodos(prev => prev.filter((_, i) => i !== index));
  };

  return (
    <View style={styles.container}>
      <Text style={styles.count}>{count}</Text>
      <View style={styles.row}>
        <Pressable style={styles.btn} onPress={() => setCount(c => c - 1)}>
          <Text style={styles.btnText}>-</Text>
        </Pressable>
        <Pressable style={styles.btn} onPress={() => setCount(c => c + 1)}>
          <Text style={styles.btnText}>+</Text>
        </Pressable>
      </View>

      <View style={styles.todoInput}>
        <TextInput
          style={styles.input}
          value={input}
          onChangeText={setInput}
          placeholder="输入待办..."
        />
        <Pressable style={styles.addBtn} onPress={addTodo}>
          <Text style={styles.btnText}>添加</Text>
        </Pressable>
      </View>

      {todos.map((todo, i) => (
        <Pressable key={i} onLongPress={() => removeTodo(i)}>
          <Text style={styles.todoItem}>• {todo}</Text>
        </Pressable>
      ))}
    </View>
  );
}
```

## 🔀 2. useReducer — 复杂组件状态

当一个组件有多个相关联的状态时，useReducer 比多个 useState 更清晰：

```tsx
import { useReducer } from 'react';

// 定义状态类型
interface FormState {
  username: string;
  email: string;
  loading: boolean;
  error: string | null;
  submitted: boolean;
}

// 定义 Action 类型
type FormAction =
  | { type: 'SET_FIELD'; field: keyof FormState; value: string }
  | { type: 'SUBMIT_START' }
  | { type: 'SUBMIT_SUCCESS' }
  | { type: 'SUBMIT_ERROR'; error: string }
  | { type: 'RESET' };

// 初始状态
const initialState: FormState = {
  username: '',
  email: '',
  loading: false,
  error: null,
  submitted: false,
};

// Reducer 函数
function formReducer(state: FormState, action: FormAction): FormState {
  switch (action.type) {
    case 'SET_FIELD':
      return { ...state, [action.field]: action.value, error: null };
    case 'SUBMIT_START':
      return { ...state, loading: true, error: null };
    case 'SUBMIT_SUCCESS':
      return { ...state, loading: false, submitted: true };
    case 'SUBMIT_ERROR':
      return { ...state, loading: false, error: action.error };
    case 'RESET':
      return initialState;
    default:
      return state;
  }
}

export default function RegisterForm() {
  const [state, dispatch] = useReducer(formReducer, initialState);

  const handleSubmit = async () => {
    dispatch({ type: 'SUBMIT_START' });
    try {
      await api.register(state.username, state.email);
      dispatch({ type: 'SUBMIT_SUCCESS' });
    } catch (e) {
      dispatch({ type: 'SUBMIT_ERROR', error: '注册失败' });
    }
  };

  return (
    <View>
      <TextInput
        value={state.username}
        onChangeText={(v) => dispatch({ type: 'SET_FIELD', field: 'username', value: v })}
        placeholder="用户名"
      />
      <TextInput
        value={state.email}
        onChangeText={(v) => dispatch({ type: 'SET_FIELD', field: 'email', value: v })}
        placeholder="邮箱"
      />
      {state.error && <Text style={{ color: 'red' }}>{state.error}</Text>}
      <Pressable onPress={handleSubmit} disabled={state.loading}>
        <Text>{state.loading ? '提交中...' : '注册'}</Text>
      </Pressable>
    </View>
  );
}
```

## 🌐 3. Context — 跨组件共享

适合全局主题、语言、用户信息等简单全局状态：

```tsx
import { createContext, useContext, useState, PropsWithChildren } from 'react';

// 1️⃣ 创建 Context
interface ThemeContextType {
  isDark: boolean;
  toggleTheme: () => void;
  colors: {
    background: string;
    text: string;
    primary: string;
  };
}

const ThemeContext = createContext<ThemeContextType | null>(null);

// 2️⃣ 创建 Provider
export function ThemeProvider({ children }: PropsWithChildren) {
  const [isDark, setIsDark] = useState(false);

  const toggleTheme = () => setIsDark(prev => !prev);

  const colors = isDark
    ? { background: '#1a1a1a', text: '#fff', primary: '#818CF8' }
    : { background: '#fff', text: '#1a1a1a', primary: '#6366F1' };

  return (
    <ThemeContext.Provider value={{ isDark, toggleTheme, colors }}>
      {children}
    </ThemeContext.Provider>
  );
}

// 3️⃣ 自定义 Hook
export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) throw new Error('useTheme must be used within ThemeProvider');
  return context;
}

// 4️⃣ 在应用入口包裹 Provider
export default function App() {
  return (
    <ThemeProvider>
      <RootLayout />
    </ThemeProvider>
  );
}

// 5️⃣ 在组件中使用
function SettingsScreen() {
  const { isDark, toggleTheme, colors } = useTheme();

  return (
    <View style={{ backgroundColor: colors.background, flex: 1, padding: 16 }}>
      <Text style={{ color: colors.text }}>当前主题：{isDark ? '深色' : '浅色'}</Text>
      <Switch value={isDark} onValueChange={toggleTheme} />
    </View>
  );
}
```

> ⚠️ **注意**：Context 的值变化时，所有消费者组件都会重新渲染。频繁变化的数据（如输入框内容）不适合用 Context。

## 🐻 4. Zustand — 轻量全局状态管理

Zustand 是当前最流行的轻量状态管理库，API 简洁、性能出色：

### 安装

```bash
npm install zustand
```

### 创建 Store

```tsx
// stores/useUserStore.ts
import { create } from 'zustand';

interface UserState {
  // 状态
  user: { id: string; name: string; avatar: string } | null;
  token: string | null;
  isLoggedIn: boolean;

  // 操作
  login: (username: string, password: string) => Promise<boolean>;
  logout: () => void;
  updateProfile: (name: string) => void;
}

export const useUserStore = create<UserState>((set, get) => ({
  user: null,
  token: null,
  isLoggedIn: false,

  login: async (username, password) => {
    try {
      const res = await api.login({ username, password });
      set({
        user: res.user,
        token: res.token,
        isLoggedIn: true,
      });
      return true;
    } catch {
      return false;
    }
  },

  logout: () => {
    set({ user: null, token: null, isLoggedIn: false });
  },

  updateProfile: (name) => {
    const { user } = get();
    if (user) {
      set({ user: { ...user, name } });
    }
  },
}));
```

### 在组件中使用

```tsx
import { useUserStore } from '../stores/useUserStore';

// 用法一：读取状态
function ProfileScreen() {
  const user = useUserStore(state => state.user);
  const logout = useUserStore(state => state.logout);

  return (
    <View>
      <Text>欢迎，{user?.name}！</Text>
      <Pressable onPress={logout}>
        <Text>退出登录</Text>
      </Pressable>
    </View>
  );
}

// 用法二：按需选取（避免不必要的重渲染）
function Header() {
  // ✅ 只订阅 name，name 不变则不重渲染
  const name = useUserStore(state => state.user?.name);
  return <Text>Hi, {name}</Text>;
}

// 用法三：在组件外使用（如工具函数、拦截器）
const token = useUserStore.getState().token;
useUserStore.getState().logout();
```

### 持久化存储

```tsx
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

export const useSettingsStore = create<SettingsState>()(
  persist(
    (set) => ({
      language: 'zh',
      fontSize: 16,
      notifications: true,

      setLanguage: (lang: string) => set({ language: lang }),
      setFontSize: (size: number) => set({ fontSize: size }),
      toggleNotifications: () =>
        set(state => ({ notifications: !state.notifications })),
    }),
    {
      name: 'settings-storage',
      storage: createJSONStorage(() => AsyncStorage),
    }
  )
);
```

## 🤔 如何选择？

```
状态类型判断流程：

这个状态只在一个组件内使用？
├── 是 → useState / useReducer
└── 否 → 需要跨组件共享
         ├── 很少变化（主题/语言）→ Context
         └── 频繁变化 / 多处使用 → Zustand
              └── 需要持久化？→ Zustand + persist 中间件
```
