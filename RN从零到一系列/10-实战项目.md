## 🐻 2. Zustand Store

```tsx
// stores/useTodoStore.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { Todo, TodoFilter, TodoCategory } from '../types/todo';

interface TodoState {
  todos: Todo[];
  filter: TodoFilter;
  searchQuery: string;

  // Actions
  addTodo: (title: string, category: TodoCategory, description?: string) => void;
  toggleTodo: (id: string) => void;
  deleteTodo: (id: string) => void;
  setFilter: (filter: TodoFilter) => void;
  setSearchQuery: (query: string) => void;

  // Computed
  filteredTodos: () => Todo[];
  stats: () => { total: number; active: number; completed: number };
}

export const useTodoStore = create<TodoState>()(
  persist(
    (set, get) => ({
      todos: [],
      filter: 'all',
      searchQuery: '',

      addTodo: (title, category, description) => {
        const newTodo: Todo = {
          id: Date.now().toString(),
          title,
          description,
          completed: false,
          createdAt: Date.now(),
          category,
        };
        set(state => ({ todos: [newTodo, ...state.todos] }));
      },

      toggleTodo: (id) => {
        set(state => ({
          todos: state.todos.map(todo =>
            todo.id === id ? { ...todo, completed: !todo.completed } : todo
          ),
        }));
      },

      deleteTodo: (id) => {
        set(state => ({
          todos: state.todos.filter(todo => todo.id !== id),
        }));
      },

      setFilter: (filter) => set({ filter }),
      setSearchQuery: (query) => set({ searchQuery: query }),

      filteredTodos: () => {
        const { todos, filter, searchQuery } = get();
        let result = todos;

        // 按状态筛选
        if (filter === 'active') result = result.filter(t => !t.completed);
        if (filter === 'completed') result = result.filter(t => t.completed);

        // 按搜索词筛选
        if (searchQuery.trim()) {
          const q = searchQuery.toLowerCase();
          result = result.filter(t =>
            t.title.toLowerCase().includes(q)
          );
        }

        return result;
      },

      stats: () => {
        const { todos } = get();
        return {
          total: todos.length,
          active: todos.filter(t => !t.completed).length,
          completed: todos.filter(t => t.completed).length,
        };
      },
    }),
    {
      name: 'todo-storage',
      storage: createJSONStorage(() => AsyncStorage),
      partialize: (state) => ({ todos: state.todos }),  // 只持久化 todos
    }
  )
);
```

```tsx
export const lightColors = {
  background: '#F9FAFB',

  primary: '#6366F1',

  background: '#111827',
  surface: '#1F2937',
  text: '#F9FAFB',
  textSecondary: '#9CA3AF',
  primary: '#818CF8',
  primaryLight: '#312E81',

  completed: '#34D399',
};

// hooks/useTheme.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { lightColors, darkColors } from '../constants/colors';

interface ThemeState {
  isDark: boolean;
  toggleTheme: () => void;
  colors: typeof lightColors;
}

export const useTheme = create<ThemeState>()(
  persist(
    (set, get) => ({
      isDark: false,
      colors: lightColors,
      toggleTheme: () => {
        const newIsDark = !get().isDark;
        set({
          isDark: newIsDark,
          colors: newIsDark ? darkColors : lightColors,
        });
      },
    }),
    {
      name: 'theme-storage',
      storage: createJSONStorage(() => AsyncStorage),
      partialize: (state) => ({ isDark: state.isDark }),
    }
  )
);
```

## 📋 4. 待办列表页

```tsx
// app/(tabs)/index.tsx
import { View, FlatList, StyleSheet, RefreshControl } from 'react-native';
import { useTodoStore } from '../../stores/useTodoStore';
import { useTheme } from '../../hooks/useTheme';
import { router } from 'expo-router';
import { TodoItem } from '../../components/TodoItem';
import { FilterBar } from '../../components/FilterBar';
import { SearchBar } from '../../components/SearchBar';
import { EmptyState } from '../../components/EmptyState';
import { FAB } from '../../components/FAB';

export default function TodoListScreen() {
  const filteredTodos = useTodoStore(s => s.filteredTodos());
  const stats = useTodoStore(s => s.stats());
  const { colors } = useTheme();
  const [refreshing, setRefreshing] = useState(false);

  const onRefresh = async () => {
    setRefreshing(true);
    // 模拟刷新
    await new Promise(resolve => setTimeout(resolve, 500));
    setRefreshing(false);
  };

  return (
    <View style={[styles.container, { backgroundColor: colors.background }]}>
      {/* 搜索栏 */}
      <SearchBar />

      {/* 筛选栏 */}
      <FilterBar stats={stats} />

      {/* 待办列表 */}
      <FlatList
        data={filteredTodos}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => <TodoItem todo={item} />}
        ListEmptyComponent={<EmptyState />}
        contentContainerStyle={styles.listContent}
        refreshControl={
          <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
        }
      />

      {/* 浮动添加按钮 */}
      <FAB onPress={() => router.push('/add')} />
    </View>
  );
}
```

## 📦 5. TodoItem 组件

```tsx
// components/TodoItem.tsx
import { View, Text, Pressable, StyleSheet, Alert } from 'react-native';
import Animated, {
  FadeIn,
  FadeOut,
  useAnimatedStyle,
  useSharedValue,
  withSpring,
} from 'react-native-reanimated';
import { Gesture, GestureDetector } from 'react-native-gesture-handler';
import { useTodoStore } from '../stores/useTodoStore';
import { useTheme } from '../hooks/useTheme';
import { Todo, CATEGORY_EMOJI } from '../types/todo';

interface Props {
  todo: Todo;
}

export const TodoItem = React.memo(function TodoItem({ todo }: Props) {
  const toggleTodo = useTodoStore(s => s.toggleTodo);
  const deleteTodo = useTodoStore(s => s.deleteTodo);
  const { colors } = useTheme();
  const translateX = useSharedValue(0);

  // 左滑删除手势
  const panGesture = Gesture.Pan()
    .activeOffsetX(-10)
    .onUpdate((e) => {
      if (e.translationX < 0) {
        translateX.value = Math.max(e.translationX, -100);
      }
    })
    .onEnd(() => {
      if (translateX.value < -60) {
        // 删除确认
        runOnJS(handleDelete)();
      }
      translateX.value = withSpring(0);
    });

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: translateX.value }],
  }));

  const handleDelete = () => {
    Alert.alert('删除确认', `确定要删除「${todo.title}」吗？`, [
      { text: '取消' },
      {
        text: '删除',
        style: 'destructive',
        onPress: () => deleteTodo(todo.id),
      },
    ]);
  };

  return (
    <Animated.View entering={FadeIn} exiting={FadeOut}>
      {/* 删除背景 */}
      <View style={[styles.deleteBackground, { backgroundColor: colors.danger }]}>
        <Text style={styles.deleteText}>🗑 删除</Text>
      </View>

      {/* 卡片主体 */}
      <GestureDetector gesture={panGesture}>
        <Animated.View
          style={[
            styles.card,
            animatedStyle,
            { backgroundColor: colors.surface, borderColor: colors.border },
          ]}
        >
          {/* 完成按钮 */}
          <Pressable
            style={[
              styles.checkbox,
              todo.completed && { backgroundColor: colors.completed, borderColor: colors.completed },
              { borderColor: colors.border },
            ]}
            onPress={() => toggleTodo(todo.id)}
          >
            {todo.completed && <Text style={styles.checkmark}>✓</Text>}
          </Pressable>

          {/* 内容 */}
          <View style={styles.content}>
            <Text
              style={[
                styles.title,
                { color: colors.text },
                todo.completed && {
                  textDecorationLine: 'line-through',
                  color: colors.textSecondary,
                },
              ]}
            >
              {CATEGORY_EMOJI[todo.category]} {todo.title}
            </Text>
            {todo.description && (
              <Text style={[styles.desc, { color: colors.textSecondary }]} numberOfLines={1}>
                {todo.description}
              </Text>
            )}
          </View>
        </Animated.View>
      </GestureDetector>
    </Animated.View>
  );
});

const styles = StyleSheet.create({
  card: {
    flexDirection: 'row',
    alignItems: 'center',
    padding: 16,
    marginHorizontal: 16,
    marginVertical: 4,
    borderRadius: 12,
    borderWidth: 1,
  },
  deleteBackground: {
    position: 'absolute',
    right: 16,
    top: 4,
    bottom: 4,
    width: 100,
    borderRadius: 12,
    justifyContent: 'center',
    alignItems: 'center',
  },
  deleteText: { color: '#fff', fontWeight: 'bold' },
  checkbox: {
    width: 24,
    height: 24,
    borderRadius: 12,
    borderWidth: 2,
    justifyContent: 'center',
    alignItems: 'center',
    marginRight: 12,
  },
  checkmark: { color: '#fff', fontSize: 14, fontWeight: 'bold' },
  content: { flex: 1 },
  title: { fontSize: 16, fontWeight: '500' },
  desc: { fontSize: 13, marginTop: 2 },
});
```

## ➕ 6. 新增待办页

```tsx
// app/add.tsx
import { View, Text, Pressable, StyleSheet, Alert } from 'react-native';
import { useForm, Controller } from 'react-hook-form';
import { router } from 'expo-router';
import { useTodoStore } from '../stores/useTodoStore';
import { useTheme } from '../hooks/useTheme';
import { TodoCategory, CATEGORY_EMOJI, CATEGORY_LABEL } from '../types/todo';

interface FormData {
  title: string;
  description: string;
  category: TodoCategory;
}

export default function AddTodoScreen() {
  const addTodo = useTodoStore(s => s.addTodo);
  const { colors } = useTheme();

  const { control, handleSubmit, setValue, watch, formState: { errors } } = useForm<FormData>({
    defaultValues: { title: '', description: '', category: 'personal' },
  });

  const selectedCategory = watch('category');

  const onSubmit = (data: FormData) => {
    addTodo(data.title, data.category, data.description || undefined);
    router.back();
  };

  return (
    <View style={[styles.container, { backgroundColor: colors.background }]}>
      <Text style={[styles.label, { color: colors.text }]}>标题 *</Text>
      <Controller
        control={control}
        name="title"
        rules={{ required: '标题不能为空', maxLength: { value: 50, message: '最多 50 字' } }}
        render={({ field: { onChange, value } }) => (
          <TextInput
            style={[styles.input, { borderColor: errors.title ? colors.danger : colors.border, color: colors.text }]}
            value={value}
            onChangeText={onChange}
            placeholder="输入待办标题..."
            placeholderTextColor={colors.textSecondary}
          />
        )}
      />
      {errors.title && <Text style={{ color: colors.danger }}>{errors.title.message}</Text>}

      <Text style={[styles.label, { color: colors.text }]}>分类</Text>
      <View style={styles.categoryRow}>
        {(['work', 'personal', 'shopping', 'health'] as TodoCategory[]).map(cat => (
          <Pressable
            key={cat}
            style={[
              styles.categoryBtn,
              {
                borderColor: selectedCategory === cat ? colors.primary : colors.border,
                backgroundColor: selectedCategory === cat ? colors.primaryLight : colors.surface,
              },
            ]}
            onPress={() => setValue('category', cat)}
          >
            <Text>{CATEGORY_EMOJI[cat]} {CATEGORY_LABEL[cat]}</Text>
          </Pressable>
        ))}
      </View>

      <Text style={[styles.label, { color: colors.text }]}>描述（可选）</Text>
      <Controller
        control={control}
        name="description"
        render={({ field: { onChange, value } }) => (
          <TextInput
            style={[styles.input, styles.textArea, { borderColor: colors.border, color: colors.text }]}
            value={value}
            onChangeText={onChange}
            placeholder="补充说明..."
            placeholderTextColor={colors.textSecondary}
            multiline
            numberOfLines={3}
          />
        )}
      />

      <Pressable
        style={[styles.submitBtn, { backgroundColor: colors.primary }]}
        onPress={handleSubmit(onSubmit)}
      >
        <Text style={styles.submitText}>添加待办</Text>
      </Pressable>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 24 },
  label: { fontSize: 14, fontWeight: '600', marginBottom: 6, marginTop: 16 },
  input: { borderWidth: 1, borderRadius: 8, padding: 12, fontSize: 16 },
  textArea: { minHeight: 80, textAlignVertical: 'top' },
  categoryRow: { flexDirection: 'row', gap: 8, flexWrap: 'wrap' },
  categoryBtn: { paddingHorizontal: 14, paddingVertical: 8, borderRadius: 20, borderWidth: 1 },
  submitBtn: { marginTop: 32, padding: 16, borderRadius: 12, alignItems: 'center' },
  submitText: { color: '#fff', fontSize: 16, fontWeight: '600' },
});
```
