## ⌨️ 1. 基础表单组件

### TextInput 进阶

```tsx
import { TextInput, View, Text, Pressable, StyleSheet, Alert } from 'react-native';
import { useState, useRef } from 'react';

export default function LoginForm() {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [showPassword, setShowPassword] = useState(false);

  // Ref 用于控制焦点
  const passwordRef = useRef<TextInput>(null);

  const handleLogin = () => {
    if (!username.trim()) return Alert.alert('提示', '请输入用户名');
    if (!password.trim()) return Alert.alert('提示', '请输入密码');
    Alert.alert('登录成功', `欢迎 ${username}！`);
  };

  return (
    <View style={styles.form}>
      <Text style={styles.label}>用户名</Text>
      <TextInput
        style={styles.input}
        placeholder="请输入用户名"
        value={username}
        onChangeText={setUsername}
        autoCapitalize="none"
        autoCorrect={false}
        returnKeyType="next"  // 键盘右下角显示"下一项"
        onSubmitEditing={() => passwordRef.current?.focus()}  // 回车跳到密码
      />

      <Text style={styles.label}>密码</Text>
      <View style={styles.passwordRow}>
        <TextInput
          ref={passwordRef}
          style={[styles.input, { flex: 1 }]}
          placeholder="请输入密码"
          value={password}
          onChangeText={setPassword}
          secureTextEntry={!showPassword}
          returnKeyType="done"
          onSubmitEditing={handleLogin}  // 回车直接登录
        />
        <Pressable onPress={() => setShowPassword(!showPassword)}>
          <Text>{showPassword ? '🙈' : '👁'}</Text>
        </Pressable>
      </View>

      <Pressable style={styles.submitBtn} onPress={handleLogin}>
        <Text style={styles.submitText}>登录</Text>
      </Pressable>
    </View>
  );
}

const styles = StyleSheet.create({
  form: { padding: 24, gap: 8 },
  label: { fontSize: 14, fontWeight: '600', color: '#374151' },
  input: {
    borderWidth: 1,
    borderColor: '#D1D5DB',
    borderRadius: 8,
    padding: 12,
    fontSize: 16,
  },
  passwordRow: {
    flexDirection: 'row',
    alignItems: 'center',
    gap: 8,
  },
  submitBtn: {
    backgroundColor: '#6366F1',
    padding: 14,
    borderRadius: 8,
    alignItems: 'center',
    marginTop: 16,
  },
  submitText: { color: '#fff', fontSize: 16, fontWeight: '600' },
});
```

### 选择器（自定义实现）

```tsx
function Picker({ options, value, onChange }) {
  return (
    <View style={styles.pickerRow}>
      {options.map(option => (
        <Pressable
          key={option.value}
          style={[
            styles.pickerItem,
            value === option.value && styles.pickerItemActive,
          ]}
          onPress={() => onChange(option.value)}
        >
          <Text style={[
            styles.pickerText,
            value === option.value && styles.pickerTextActive,
          ]}>
            {option.label}
          </Text>
        </Pressable>
      ))}
    </View>
  );
}

// 使用

  options={[
    { label: '男', value: 'male' },
```

## 🛡 2. 表单验证

### 手动验证

```tsx
interface FormErrors {
  username?: string;
  email?: string;
  password?: string;
}

const validate = (values: FormValues): FormErrors => {
  const errors: FormErrors = {};

  if (!values.username.trim()) {
    errors.username = '用户名不能为空';
  } else if (values.username.length < 3) {
    errors.username = '用户名至少 3 个字符';
  }

  if (!values.email.trim()) {
    errors.email = '邮箱不能为空';
  } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(values.email)) {
    errors.email = '邮箱格式不正确';
  }

  if (!values.password) {
    errors.password = '密码不能为空';
  } else if (values.password.length < 6) {
    errors.password = '密码至少 6 位';
  }

  return errors;
};

// 错误提示组件
function ErrorText({ message }: { message?: string }) {
  if (!message) return null;
  return <Text style={{ color: '#EF4444', fontSize: 12, marginTop: 4 }}>{message}</Text>;
}
```

### react-hook-form

推荐使用 react-hook-form 管理复杂表单：

```bash
npm install react-hook-form
```

```tsx
import { useForm, Controller } from 'react-hook-form';

interface RegisterForm {
  username: string;
  email: string;
  password: string;
  confirmPassword: string;
}

export default function RegisterScreen() {
  const { control, handleSubmit, watch, formState: { errors } } = useForm<RegisterForm>({
    defaultValues: {
      username: '',
      email: '',
      password: '',
      confirmPassword: '',
    },
  });

  const password = watch('password');

  const onSubmit = (data: RegisterForm) => {
    console.log('提交:', data);
  };

  return (
    <ScrollView contentContainerStyle={styles.form}>
      {/* 用户名 */}
      <Text style={styles.label}>用户名</Text>
      <Controller
        control={control}
        name="username"
        rules={{
          required: '用户名不能为空',
          minLength: { value: 3, message: '至少 3 个字符' },
          maxLength: { value: 20, message: '最多 20 个字符' },
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <TextInput
            style={[styles.input, errors.username && styles.inputError]}
            onBlur={onBlur}
            onChangeText={onChange}
            value={value}
            placeholder="请输入用户名"
          />
        )}
      />
      <ErrorText message={errors.username?.message} />

      {/* 邮箱 */}
      <Text style={styles.label}>邮箱</Text>
      <Controller
        control={control}
        name="email"
        rules={{
          required: '邮箱不能为空',
          pattern: {
            value: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
            message: '邮箱格式不正确',
          },
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <TextInput
            style={[styles.input, errors.email && styles.inputError]}
            onBlur={onBlur}
            onChangeText={onChange}
            value={value}
            placeholder="请输入邮箱"
            keyboardType="email-address"
          />
        )}
      />
      <ErrorText message={errors.email?.message} />

      {/* 密码 */}
      <Text style={styles.label}>密码</Text>
      <Controller
        control={control}
        name="password"
        rules={{
          required: '密码不能为空',
          minLength: { value: 6, message: '至少 6 位' },
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <TextInput
            style={[styles.input, errors.password && styles.inputError]}
            onBlur={onBlur}
            onChangeText={onChange}
            value={value}
            placeholder="请输入密码"
            secureTextEntry
          />
        )}
      />
      <ErrorText message={errors.password?.message} />

      {/* 确认密码 */}
      <Text style={styles.label}>确认密码</Text>
      <Controller
        control={control}
        name="confirmPassword"
        rules={{
          required: '请确认密码',
          validate: (value) => value === password || '两次密码不一致',
        }}
        render={({ field: { onChange, onBlur, value } }) => (
          <TextInput
            style={[styles.input, errors.confirmPassword && styles.inputError]}
            onBlur={onBlur}
            onChangeText={onChange}
            value={value}
            placeholder="请再次输入密码"
            secureTextEntry
          />
        )}
      />
      <ErrorText message={errors.confirmPassword?.message} />

      <Pressable style={styles.submitBtn} onPress={handleSubmit(onSubmit)}>
        <Text style={styles.submitText}>注册</Text>
      </Pressable>
    </ScrollView>
  );
}
```

## ⌨️ 3. 键盘适配

在移动端，键盘弹出会遮挡输入框，需要做适配：

### KeyboardAvoidingView

```tsx
import { KeyboardAvoidingView, Platform, ScrollView } from 'react-native';

export default function FormScreen() {
  return (
    <KeyboardAvoidingView
      style={{ flex: 1 }}
      behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
      keyboardVerticalOffset={Platform.OS === 'ios' ? 64 : 0}
    >
      <ScrollView
        contentContainerStyle={styles.form}
        keyboardShouldPersistTaps="handled"  // 点击空白处不收起键盘
      >
        {/* 表单字段 */}
      </ScrollView>
    </KeyboardAvoidingView>
  );
}
```

### 收起键盘

```tsx
import { Keyboard, TouchableWithoutFeedback } from 'react-native';

// 方式一：点击空白处收起
<TouchableWithoutFeedback onPress={Keyboard.dismiss}>
  <View style={{ flex: 1 }}>
    {/* 内容 */}
  </View>
</TouchableWithoutFeedback>

// 方式二：代码手动收起
Keyboard.dismiss();

// 方式三：监听键盘事件
useEffect(() => {
  const showSub = Keyboard.addListener('keyboardDidShow', (e) => {
    console.log('键盘高度:', e.endCoordinates.height);
  });
  const hideSub = Keyboard.addListener('keyboardDidHide', () => {
    console.log('键盘已隐藏');
  });

  return () => {
    showSub.remove();
    hideSub.remove();
  };
}, []);
```
