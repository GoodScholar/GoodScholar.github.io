## 💾 1. AsyncStorage — 本地存储

相当于 Web 的 localStorage，用于持久化简单数据：

### 安装

```bash
npx expo install @react-native-async-storage/async-storage
```

### 基本用法

```tsx
import AsyncStorage from '@react-native-async-storage/async-storage';

// 存储字符串
await AsyncStorage.setItem('username', '张三');

// 读取字符串
const username = await AsyncStorage.getItem('username');

// 存储对象（需要 JSON 序列化）
const user = { id: '1', name: '张三', age: 25 };
await AsyncStorage.setItem('user', JSON.stringify(user));

// 读取对象
const userStr = await AsyncStorage.getItem('user');
const userData = userStr ? JSON.parse(userStr) : null;

// 删除
await AsyncStorage.removeItem('username');

// 清空全部
await AsyncStorage.clear();

// 获取所有 key
const keys = await AsyncStorage.getAllKeys();

// 批量操作
await AsyncStorage.multiSet([
  ['key1', 'value1'],
  ['key2', 'value2'],
]);
```

### 封装 Storage 工具

```tsx
// utils/storage.ts
import AsyncStorage from '@react-native-async-storage/async-storage';

export const storage = {
  async get<T>(key: string): Promise<T | null> {
    try {
      const value = await AsyncStorage.getItem(key);
      return value ? JSON.parse(value) : null;
    } catch {
      return null;
    }
  },

  async set(key: string, value: unknown): Promise<void> {
    try {
      await AsyncStorage.setItem(key, JSON.stringify(value));
    } catch (error) {
      console.error('Storage set error:', error);
    }
  },

  async remove(key: string): Promise<void> {
    await AsyncStorage.removeItem(key);
  },

  async clear(): Promise<void> {
    await AsyncStorage.clear();
  },
};

// 使用
await storage.set('token', 'abc123');
const token = await storage.get<string>('token');
```

## 📷 2. 相机与相册

### 安装

```bash
npx expo install expo-image-picker
```

### 选择图片

```tsx
import * as ImagePicker from 'expo-image-picker';

export default function AvatarPicker() {
  const [image, setImage] = useState<string | null>(null);

  // 从相册选择
  const pickImage = async () => {
    // 请求权限
    const { status } = await ImagePicker.requestMediaLibraryPermissionsAsync();
    if (status !== 'granted') {
      Alert.alert('提示', '需要相册权限才能选择图片');
      return;
    }

    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ['images'],
      allowsEditing: true,          // 允许裁剪
      aspect: [1, 1],               // 裁剪比例 1:1
      quality: 0.8,                 // 压缩质量
    });

    if (!result.canceled) {
      setImage(result.assets[0].uri);
    }
  };

  // 拍照
  const takePhoto = async () => {
    const { status } = await ImagePicker.requestCameraPermissionsAsync();
    if (status !== 'granted') {
      Alert.alert('提示', '需要相机权限');
      return;
    }

    const result = await ImagePicker.launchCameraAsync({
      allowsEditing: true,
      aspect: [1, 1],
      quality: 0.8,
    });

    if (!result.canceled) {
      setImage(result.assets[0].uri);
    }
  };

  return (
    <View style={styles.container}>
      <Pressable onPress={pickImage}>
        {image ? (
          <Image source={{ uri: image }} style={styles.avatar} />
        ) : (
          <View style={styles.placeholder}>
            <Text>+</Text>
          </View>
        )}
      </Pressable>
      <Pressable onPress={takePhoto}>
        <Text>拍照</Text>
      </Pressable>
    </View>
  );
}
```

## 📍 3. 地理定位

### 安装

```bash
npx expo install expo-location
```

### 获取位置

```tsx
import * as Location from 'expo-location';

export default function LocationDemo() {
  const [location, setLocation] = useState<Location.LocationObject | null>(null);
  const [address, setAddress] = useState('');

  const getLocation = async () => {
    // 请求权限
    const { status } = await Location.requestForegroundPermissionsAsync();
    if (status !== 'granted') {
      Alert.alert('提示', '需要定位权限');
      return;
    }

    // 获取当前位置
    const loc = await Location.getCurrentPositionAsync({
      accuracy: Location.Accuracy.High,
    });
    setLocation(loc);

    // 反地理编码（坐标 → 地址）
    const [addr] = await Location.reverseGeocodeAsync({
      latitude: loc.coords.latitude,
      longitude: loc.coords.longitude,
    });
    if (addr) {
      setAddress(`${addr.city} ${addr.district} ${addr.street}`);
    }
  };

  return (
    <View style={styles.container}>
      <Pressable style={styles.btn} onPress={getLocation}>
        <Text style={styles.btnText}>获取位置</Text>
      </Pressable>
      {location && (
        <View>
          <Text>纬度: {location.coords.latitude.toFixed(6)}</Text>
          <Text>经度: {location.coords.longitude.toFixed(6)}</Text>
          <Text>地址: {address}</Text>
        </View>
      )}
    </View>
  );
}
```

## 📤 4. 系统分享

React Native 内置 Share API，无需安装额外库：

```tsx
import { Share } from 'react-native';

const onShare = async () => {
  try {
    const result = await Share.share({
      title: '分享标题',
      message: '来看看这个有趣的链接 https://example.com',
      // iOS 支持 url 字段
      url: 'https://example.com',
    });

    if (result.action === Share.sharedAction) {
      console.log('分享成功');
    } else if (result.action === Share.dismissedAction) {
      console.log('取消分享');
    }
  } catch (error) {
    Alert.alert('分享失败');
  }
};
```

## 🔐 6. 生物识别

```bash
npx expo install expo-local-authentication
```

```tsx
import * as LocalAuthentication from 'expo-local-authentication';

const authenticate = async () => {
  // 检查硬件支持
  const hasHardware = await LocalAuthentication.hasHardwareAsync();
  if (!hasHardware) {
    Alert.alert('提示', '设备不支持生物识别');
    return;
  }

  // 检查是否已录入
  const isEnrolled = await LocalAuthentication.isEnrolledAsync();
  if (!isEnrolled) {
    Alert.alert('提示', '请先在系统设置中录入指纹/面容');
    return;
  }

  // 发起认证
  const result = await LocalAuthentication.authenticateAsync({
    promptMessage: '请验证身份',
    cancelLabel: '取消',
    fallbackLabel: '使用密码',
  });

  if (result.success) {
    Alert.alert('验证成功！');
  } else {
    Alert.alert('验证失败');
  }
};
```
