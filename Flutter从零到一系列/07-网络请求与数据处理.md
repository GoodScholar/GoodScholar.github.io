## 🛠 1. Dio — 网络请求库

### 安装

```yaml
dependencies:
  dio: ^5.7.0
```

### 基础用法

```dart
import 'package:dio/dio.dart';

final dio = Dio(BaseOptions(
  baseUrl: 'https://api.example.com',
  connectTimeout: const Duration(seconds: 10),
  receiveTimeout: const Duration(seconds: 10),
  headers: {'Content-Type': 'application/json'},
));

// GET 请求
Future<void> getProducts() async {
  try {
    final response = await dio.get('/api/products', queryParameters: {'page': 1});
    print(response.data);  // Map / List
    print(response.statusCode);  // 200
  } on DioException catch (e) {
    print('请求失败: ${e.message}');
  }
}

// POST 请求
Future<void> login(String email, String password) async {
  try {
    final response = await dio.post('/api/auth/login', data: {
      'email': email,
      'password': password,
    });
    final token = response.data['token'];
    print('登录成功，Token: $token');
  } on DioException catch (e) {
    print('登录失败: ${e.response?.data}');
  }
}

// PUT 请求
Future<void> updateProfile(String name) async {
  await dio.put('/api/profile', data: {'name': name});
}

// DELETE 请求
Future<void> deleteItem(String id) async {
  await dio.delete('/api/items/$id');
}
```

### 拦截器

```dart
// 封装带拦截器的 Dio
class HttpClient {
  static final Dio _dio = Dio(BaseOptions(
    baseUrl: 'https://api.example.com',
    connectTimeout: const Duration(seconds: 10),
  ))..interceptors.addAll([
    // 请求拦截器 — 自动添加 Token
    InterceptorsWrapper(
      onRequest: (options, handler) {
        final token = getToken();  // 从本地存储获取
        if (token != null) {
          options.headers['Authorization'] = 'Bearer $token';
        }
        print('→ ${options.method} ${options.uri}');
        handler.next(options);
      },
      onResponse: (response, handler) {
        print('← ${response.statusCode} ${response.requestOptions.uri}');
        handler.next(response);
      },
      onError: (error, handler) {
        if (error.response?.statusCode == 401) {
          // Token 过期，跳转登录页
          navigateToLogin();
        }
        handler.next(error);
      },
    ),

    // 日志拦截器（开发环境使用）
    LogInterceptor(requestBody: true, responseBody: true),
  ]);

  static Dio get instance => _dio;
}
```

### 定义模型

```dart
// lib/models/product.dart
import 'package:freezed_annotation/freezed_annotation.dart';

part 'product.freezed.dart';
part 'product.g.dart';

@freezed
class Product with _$Product {
  const factory Product({
    required String id,
    required String name,
    required double price,
    @Default('') String image,
    @Default('') String description,
    @Default(0) int sales,
    @JsonKey(name: 'created_at') DateTime? createdAt,
  }) = _Product;

  factory Product.fromJson(Map<String, dynamic> json) =>
      _$ProductFromJson(json);
}

@freezed
class ApiResponse<T> with _$ApiResponse<T> {
  const factory ApiResponse({
    required int code,
    required String message,
    T? data,
  }) = _ApiResponse;

  factory ApiResponse.fromJson(
    Map<String, dynamic> json,
    T Function(Object?) fromJsonT,
  ) => _$ApiResponseFromJson(json, fromJsonT);
}
```


### 使用模型

```dart
// JSON → 对象
final json = {'id': '1', 'name': 'Flutter', 'price': 59.0};
```

## 🔗 3. 完整的网络层封装

### API 服务层

```dart
// lib/api/product_api.dart
class ProductApi {
  final Dio _dio;
  ProductApi(this._dio);

  Future<List<Product>> getProducts({int page = 1, int pageSize = 20}) async {
    final response = await _dio.get('/api/products', queryParameters: {
      'page': page,
      'pageSize': pageSize,
    });
    final list = response.data['data']['list'] as List;
    return list.map((e) => Product.fromJson(e)).toList();
  }

  Future<Product> getProductDetail(String id) async {
    final response = await _dio.get('/api/products/$id');
    return Product.fromJson(response.data['data']);
  }

  Future<void> addToCart(String productId, int quantity) async {
    await _dio.post('/api/cart', data: {
      'productId': productId,
      'quantity': quantity,
    });
  }
}
```

### 配合 Riverpod 使用

```dart
// Provider 注册
@riverpod
Dio dio(Ref ref) => HttpClient.instance;

@riverpod
ProductApi productApi(Ref ref) => ProductApi(ref.watch(dioProvider));

// 数据 Provider
@riverpod
Future<List<Product>> productList(Ref ref) async {
  final api = ref.watch(productApiProvider);
  return api.getProducts();
}

@riverpod
Future<Product> productDetail(Ref ref, {required String id}) async {
  final api = ref.watch(productApiProvider);
  return api.getProductDetail(id);
}
```

## 🛡 4. 错误处理

```dart
// 统一异常类
sealed class AppException implements Exception {
  String get message;
}

class NetworkException extends AppException {
  @override
  final String message;
  final int? statusCode;
  NetworkException({required this.message, this.statusCode});
}

class ServerException extends AppException {
  @override
  final String message;
  ServerException({required this.message});
}

// Dio 错误转化
AppException handleDioError(DioException e) {
  switch (e.type) {
    case DioExceptionType.connectionTimeout:
    case DioExceptionType.sendTimeout:
    case DioExceptionType.receiveTimeout:
      return NetworkException(message: '网络连接超时，请检查网络');
    case DioExceptionType.connectionError:
      return NetworkException(message: '网络连接失败');
    case DioExceptionType.badResponse:
      final statusCode = e.response?.statusCode;
      final msg = e.response?.data?['message'] ?? '服务器错误';
      return ServerException(message: '$msg ($statusCode)');
    default:
      return NetworkException(message: '未知网络错误');
  }
}

// 在 API 层使用
Future<List<Product>> getProducts() async {
  try {
    final response = await _dio.get('/api/products');
    return (response.data['list'] as List).map((e) => Product.fromJson(e)).toList();
  } on DioException catch (e) {
    throw handleDioError(e);
  }
}

// UI 层展示错误
productsAsync.when(
  data: (products) => ProductListView(products: products),
  loading: () => CircularProgressIndicator(),
  error: (e, _) => Text(
    e is AppException ? e.message : '未知错误',
  ),
);
```
