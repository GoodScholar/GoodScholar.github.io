# 📜 Flutter 从零到一（八）：列表与滚动 — 玩转大数据量展示

> **系列导读**：几乎每个 App 都有列表。本文从基础 ListView 到高级 Sliver，
> 教你实现分页加载、下拉刷新、瀑布流等常见需求。

## 📊 列表 Widget 速查

| Widget | 用途 | 特点 |
|--------|------|------|
| `ListView` | 线性列表 | 最常用 |
| `ListView.builder` | 大数据列表 | 懒加载，按需创建 |
| `ListView.separated` | 带分割线列表 | 自动添加分隔 |
| `GridView` | 网格布局 | 多列展示 |
| `CustomScrollView` | 组合滚动 | 灵活组合 Sliver |
| `PageView` | 整页滚动 | 引导页/轮播 |

## 📋 1. ListView 基础

### 静态列表

```dart
ListView(
  padding: EdgeInsets.all(16),

    ListTile(leading: Icon(Icons.email), title: Text('邮件'), trailing: Text('99+')),

```

### 动态列表（大数据量必用）

```dart
ListView.builder(
  itemCount: products.length,
  itemBuilder: (context, index) {
    final product = products[index];
    return Card(
      margin: EdgeInsets.symmetric(horizontal: 16, vertical: 6),
      child: ListTile(
        leading: ClipRRect(
          borderRadius: BorderRadius.circular(8),
          child: Image.network(product.image, width: 60, height: 60, fit: BoxFit.cover),
        ),
        title: Text(product.name, maxLines: 1, overflow: TextOverflow.ellipsis),
        subtitle: Text('¥${product.price}', style: TextStyle(color: Colors.red)),
        trailing: Icon(Icons.chevron_right),
        onTap: () => context.push('/product/${product.id}'),
      ),
    );
  },
)
```

### 带分割线列表

```dart
ListView.separated(
  itemCount: items.length,
  separatorBuilder: (_, __) => Divider(height: 1, indent: 72),
  itemBuilder: (context, index) {
    return ListTile(
      leading: CircleAvatar(child: Text('${index + 1}')),
      title: Text(items[index]),
    );
  },
)
```

## 🔄 2. 下拉刷新 + 上拉加载

```dart
class ProductListPage extends ConsumerStatefulWidget {
  const ProductListPage({super.key});
  @override
  ConsumerState<ProductListPage> createState() => _ProductListPageState();
}

class _ProductListPageState extends ConsumerState<ProductListPage> {
  final _scrollController = ScrollController();
  bool _isLoadingMore = false;
  int _page = 1;

  @override
  void initState() {
    super.initState();
    // 监听滚动到底部
    _scrollController.addListener(() {
      if (_scrollController.position.pixels >=
          _scrollController.position.maxScrollExtent - 100) {
        _loadMore();
      }
    });
  }

  @override
  void dispose() {
    _scrollController.dispose();
    super.dispose();
  }

  Future<void> _loadMore() async {
    if (_isLoadingMore) return;
    setState(() => _isLoadingMore = true);
    _page++;
    await ref.read(productListProvider.notifier).loadMore(_page);
    setState(() => _isLoadingMore = false);
  }

  Future<void> _refresh() async {
    _page = 1;
    await ref.read(productListProvider.notifier).refresh();
  }

  @override
  Widget build(BuildContext context) {
    final productsAsync = ref.watch(productListProvider);

    return Scaffold(
      appBar: AppBar(title: Text('商品列表')),
      body: productsAsync.when(
        data: (products) => RefreshIndicator(
          onRefresh: _refresh,
          child: ListView.builder(
            controller: _scrollController,
            itemCount: products.length + 1,  // +1 给加载指示器
            itemBuilder: (context, index) {
              if (index == products.length) {
                return _isLoadingMore
                    ? Padding(
                        padding: EdgeInsets.all(16),
                        child: Center(child: CircularProgressIndicator()),
                      )
                    : SizedBox.shrink();
              }
              return ProductCard(product: products[index]);
            },
          ),
        ),
        loading: () => Center(child: CircularProgressIndicator()),
        error: (e, _) => Center(child: Text('加载失败')),
      ),
    );
  }
}
```

## 🧩 4. CustomScrollView + Sliver（高级组合）

```dart
// 吸顶搜索栏 + 分类横滑 + 商品网格 — 一个页面多种滚动区域
class HomePage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: CustomScrollView(
        slivers: [
          // 可折叠的 AppBar
          SliverAppBar(
            expandedHeight: 200,
            pinned: true,
            flexibleSpace: FlexibleSpaceBar(
              title: Text('发现好物'),
              background: Image.network(
                'https://picsum.photos/400/200',
                fit: BoxFit.cover,
              ),
            ),
          ),

          // 搜索栏（固定在顶部）
          SliverPersistentHeader(
            pinned: true,
            delegate: _SearchBarDelegate(),
          ),

          // 标题
          SliverToBoxAdapter(
            child: Padding(
              padding: EdgeInsets.all(16),
              child: Text('热门推荐', style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold)),
            ),
          ),

          // 商品网格
          SliverGrid(
            delegate: SliverChildBuilderDelegate(
              (context, index) => ProductGridCard(product: products[index]),
              childCount: products.length,
            ),
            gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
              crossAxisCount: 2,
              crossAxisSpacing: 12,
              mainAxisSpacing: 12,
              childAspectRatio: 0.75,
            ),
          ),

          // 底部间距
          SliverToBoxAdapter(
            child: SizedBox(height: 100),
          ),
        ],
      ),
    );
  }
}
```

## 📖 5. PageView — 整页滑动

```dart
// 引导页
class OnboardingPage extends StatefulWidget {
  @override
  State<OnboardingPage> createState() => _OnboardingPageState();
}

class _OnboardingPageState extends State<OnboardingPage> {
  final _controller = PageController();
  int _currentPage = 0;

  final _pages = [
    {'icon': Icons.explore, 'title': '发现新世界', 'desc': '探索海量优质内容'},
    {'icon': Icons.favorite, 'title': '收藏喜欢的', 'desc': '一键收藏感兴趣的内容'},
    {'icon': Icons.share, 'title': '分享给朋友', 'desc': '好东西要和朋友一起分享'},
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Column(
        children: [
          Expanded(
            child: PageView.builder(
              controller: _controller,
              onPageChanged: (i) => setState(() => _currentPage = i),
              itemCount: _pages.length,
              itemBuilder: (_, index) {
                final page = _pages[index];
                return Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    Icon(page['icon'] as IconData, size: 100, color: Colors.indigo),
                    SizedBox(height: 32),
                    Text(page['title'] as String, style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold)),
                    SizedBox(height: 12),
                    Text(page['desc'] as String, style: TextStyle(color: Colors.grey)),
                  ],
                );
              },
            ),
          ),
          // 指示器
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: List.generate(3, (i) => Container(
              width: _currentPage == i ? 24 : 8,
              height: 8,
              margin: EdgeInsets.symmetric(horizontal: 4),
              decoration: BoxDecoration(
                color: _currentPage == i ? Colors.indigo : Colors.grey[300],
                borderRadius: BorderRadius.circular(4),
              ),
            )),
          ),
          SizedBox(height: 32),
          if (_currentPage == 2)
            Padding(
              padding: EdgeInsets.symmetric(horizontal: 32),
              child: SizedBox(
                width: double.infinity,
                child: FilledButton(
                  onPressed: () => context.go('/home'),
                  child: Text('开始使用'),
                ),
              ),
            ),
          SizedBox(height: 48),
        ],
      ),
    );
  }
}
```
